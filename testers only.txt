        firePart.CanQuery = false
        firePart.CanTouch = false

        firePart.Parent = workspace

        local spread = Vector3.new(math.random(-30, 30), math.random(-10, 10), math.random(-30, 30))
        firePart.AssemblyLinearVelocity = (direction * 150) + spread
        game:GetService("Debris"):AddItem(firePart, 0.8)
    end)
end

-- ========== RAINBOW ORBS ==========
local function spawnRainbowOrb(pos)
    pcall(function()
        -- Get the gun muzzle position
        local char = game.Players.LocalPlayer.Character
        local tool = char and char:FindFirstChildOfClass("Tool")
        local startPos
        if tool and tool:FindFirstChild("Handle") then
            local handle = tool.Handle
            local muzzle = handle:FindFirstChild("GunMuzzlePoint") or handle:FindFirstChild("GunFirePoint")
            if muzzle then
                startPos = muzzle.WorldCFrame.Position
            else
                startPos = handle.Position
            end
        else
            startPos = pos
        end

        -- Direction from gun to target
        local direction = (pos - startPos).Unit
        local orb = Instance.new("Part")
        orb.Anchored = false
        orb.CanCollide = false
        orb.CanQuery = false
        orb.CanTouch = false
        orb.Material = Enum.Material.Neon
        orb.Shape = Enum.PartType.Ball
        orb.Size = Vector3.new(4, 4, 4)
        orb.CFrame = CFrame.new(startPos)
        orb.Color = Color3.fromHSV(math.random(), 1, 1)
        orb.Parent = workspace
        -- Fly forward from the gun
        orb.AssemblyLinearVelocity = direction * 200
        game:GetService("Debris"):AddItem(orb, 1.5)
    end)
end

local UIS_LB = game:GetService("UserInputService")
UIS_LB.InputBegan:Connect(function(input, gp)
    if gp then return end
    if not getgenv().lightningBullets and not getgenv().rainbowOrbs and not getgenv().flameBullets then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
        local char = game.Players.LocalPlayer.Character
        if not char or not char:FindFirstChildOfClass("Tool") then return end
        if tick() - getgenv().lightningCooldown < 0.05 then return end
        getgenv().lightningCooldown = tick()
        local mouse = game.Players.LocalPlayer:GetMouse()
        if mouse and mouse.Hit then
            local tool = char:FindFirstChildOfClass("Tool")
            local startPos = mouse.Hit.Position
            local direction = mouse.Hit.LookVector
            if tool and tool:FindFirstChild("Handle") then
                local handle = tool.Handle
                local muzzle = handle:FindFirstChild("GunMuzzlePoint") or handle:FindFirstChild("GunFirePoint")
                if muzzle then
                    startPos = muzzle.WorldCFrame.Position
                else
                    startPos = handle.Position
                end
                direction = (mouse.Hit.Position - startPos).Unit
            end

            if getgenv().lightningBullets then
                task.spawn(spawnLightning, mouse.Hit.Position)
            end
            if getgenv().rainbowOrbs then
                task.spawn(spawnRainbowOrb, mouse.Hit.Position)
            end
            if getgenv().flameBullets then
                -- Spawn 5 fire balls per shot
                for i = 1, 5 do
                    task.spawn(spawnFire, startPos, direction)
                end
            end
        end
    end
end)

-- ========== AUTO DUPE ==========
local AutoDupeEnabled = false
local DupeCooldown = false

local function getPing()
    local ok, result = pcall(function()
        return tonumber(string.match(game.Players.LocalPlayer:GetNetworkPing(), "%d+"))
    end)
    if ok and result then return result end
    return 100
end

local function duplicateItem()
    if DupeCooldown then return end
    DupeCooldown = true

    local lp = game.Players.LocalPlayer
    local rs = game:GetService("ReplicatedStorage")
    local character = lp.Character
    local backpack = lp:FindFirstChild("Backpack")
    if not character or not backpack then
        DupeCooldown = false
        return
    end

    local tool = character:FindFirstChildOfClass("Tool")
    if not tool then
        DupeCooldown = false
        return
    end

    local toolName = tool.Name
    local toolId = nil

    -- Move tool to backpack
    tool.Parent = backpack
    task.wait(0.5)

    -- Set up listener for market item
    local connection
    if rs:FindFirstChild("MarketItems") then
        connection = rs.MarketItems.ChildAdded:Connect(function(item)
            if item.Name == toolName then
                local owner = item:WaitForChild("owner", 2)
                if owner and owner.Value == lp.Name then
                    toolId = item:GetAttribute("SpecialId")
                end
            end
        end)
    end

    -- List the weapon
    task.spawn(function()
        if rs:FindFirstChild("ListWeaponRemote") then
            rs.ListWeaponRemote:FireServer(toolName, 676767)
        end
    end)

    local ping = getPing()
    local delayTime = 0.25 + ((math.clamp(ping, 0, 300) / 300) * 0.03)
    task.wait(delayTime)

    -- Store in backpack
    task.spawn(function()
        if rs:FindFirstChild("BackpackRemote") then
            rs.BackpackRemote:InvokeServer("Store", toolName)
        end
    end)

    task.wait(3)

    -- Remove from market and grab back
    if toolId then
        task.spawn(function()
            if rs:FindFirstChild("BuyItemRemote") then
                rs.BuyItemRemote:FireServer(toolName, "Remove", toolId)
            end
        end)
    end

    task.spawn(function()
        if rs:FindFirstChild("BackpackRemote") then
            rs.BackpackRemote:InvokeServer("Grab", toolName)
        end
    end)

    task.wait(1)

    -- Re-equip
    local dupedTool = backpack:FindFirstChild(toolName)
    if dupedTool then
        dupedTool.Parent = character
    end

    if connection then
        connection:Disconnect()
    end

    task.wait(2)
    DupeCooldown = false
end

task.spawn(function()
    while task.wait(0.5) do
        if AutoDupeEnabled and not DupeCooldown then
            pcall(duplicateItem)
        end
    end
end)

-- ========== Keep Tools on Death (from one to keep for ui) ==========
local KeepToolsOnDeath = false
local MarketProtectedTools = {}
local MarketHealthThreshold = 35

local ReplicatedStorage_KTD = game:GetService("ReplicatedStorage")
local LocalPlayer_KTD = game.Players.LocalPlayer

local function FindMarketToolId(ToolName)
    if not ReplicatedStorage_KTD:FindFirstChild("MarketItems") then return nil end
    for _, Item in ipairs(ReplicatedStorage_KTD.MarketItems:GetChildren()) do
        if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer_KTD.Name then
            return Item:GetAttribute("SpecialId") or Item:GetAttribute("ToolId")
        end
    end
end

local function MarketProtectTool(ToolName)
    if MarketProtectedTools[ToolName] then return end
    local ToolId
    local Connection
    if ReplicatedStorage_KTD:FindFirstChild("MarketItems") then
        Connection = ReplicatedStorage_KTD.MarketItems.ChildAdded:Connect(function(Item)
            if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer_KTD.Name then
                ToolId = Item:GetAttribute("SpecialId")
            end
        end)
    end
    pcall(function()
        if LocalPlayer_KTD.Character and LocalPlayer_KTD.Character:FindFirstChild("Humanoid") then
            LocalPlayer_KTD.Character:FindFirstChild("Humanoid"):UnequipTools()
        end
    end)
    task.wait(0.15)
    pcall(function()
        if ReplicatedStorage_KTD:FindFirstChild("ListWeaponRemote") then
            ReplicatedStorage_KTD.ListWeaponRemote:FireServer(ToolName, 676767)
        end
    end)
    task.wait(0.3)
    pcall(function()
        if ReplicatedStorage_KTD:FindFirstChild("BackpackRemote") then
            ReplicatedStorage_KTD.BackpackRemote:InvokeServer("Store", ToolName)
        end
    end)
    task.wait(3)
    if not ToolId then ToolId = FindMarketToolId(ToolName) end
    if Connection then Connection:Disconnect() end
    if ToolId then MarketProtectedTools[ToolName] = ToolId end
end

local function MarketReleaseTool(ToolName, ToolId)
    if not ToolName or not ToolId then return end
    if not ReplicatedStorage_KTD:FindFirstChild("BuyItemRemote") or not ReplicatedStorage_KTD:FindFirstChild("BackpackRemote") then return end
    pcall(function() ReplicatedStorage_KTD.BuyItemRemote:FireServer(ToolName, "Remove", ToolId) end)
    task.wait(0.5)
    pcall(function() ReplicatedStorage_KTD.BackpackRemote:InvokeServer("Grab", ToolName) end)
    MarketProtectedTools[ToolName] = nil
end

local function ProtectInventoryTools()
    for _, ToolName in ipairs((function()
        local names = {}
        if LocalPlayer_KTD.Character then
            local tool = LocalPlayer_KTD.Character:FindFirstChildOfClass("Tool")
            if tool then table.insert(names, tool.Name) end
        end
        return names
    end)()) do
        MarketProtectTool(ToolName)
    end
end

local function ReleaseMarketTools()
    for ToolName, ToolId in pairs(MarketProtectedTools) do
        MarketReleaseTool(ToolName, ToolId)
    end
end

local function SetupKeepTools(character)
    if not character then return end
    local humanoid = character:WaitForChild("Humanoid")
    humanoid.Died:Connect(function()
        if KeepToolsOnDeath then
            task.spawn(ProtectInventoryTools)
        end
    end)
    humanoid.HealthChanged:Connect(function(value)
        if not KeepToolsOnDeath then return end
        if value > 0 and value <= MarketHealthThreshold then
            task.spawn(ProtectInventoryTools)
        elseif value > MarketHealthThreshold and next(MarketProtectedTools) then
            task.spawn(ReleaseMarketTools)
        end
    end)
end

if LocalPlayer_KTD.Character then
    SetupKeepTools(LocalPlayer_KTD.Character)
end
LocalPlayer_KTD.CharacterAdded:Connect(function(char)
    SetupKeepTools(char)
    task.wait(0.5)
    if KeepToolsOnDeath and LocalPlayer_KTD.Character then
        local humanoid = LocalPlayer_KTD.Character:FindFirstChildOfClass("Humanoid")
        if humanoid and humanoid.Health > MarketHealthThreshold and next(MarketProtectedTools) then
            task.spawn(ReleaseMarketTools)
        end
    end
end)

local function handleSpeed()
    if not swimMethod then return end
    
    local character = player.Character
    if not character or not character.Parent then return end
    
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not hrp or humanoid.Health <= 0 then return end
    
    humanoid:ChangeState(0)

    if flyEnabled then
        hrp.RotVelocity = Vector3.new(0, 0, 0)
        return
    end
    
    local moveDir = humanoid.MoveDirection
    if moveDir.Magnitude > 0 then
        local worldMoveVec = Vector3.new(moveDir.X, 0, moveDir.Z).Unit * (50 * 16)
        if worldMoveVec.Magnitude == worldMoveVec.Magnitude then 
            hrp.Velocity = Vector3.new(worldMoveVec.X, hrp.Velocity.Y, worldMoveVec.Z)
        end
    else
        hrp.Velocity = Vector3.new(0, hrp.Velocity.Y, 0)
    end
    
    hrp.RotVelocity = Vector3.new(0, 0, 0)
end

    local success = pcall(function()

        char:WaitForChild("HumanoidRootPart", 5)
    end)
    
    if not success then return end
    
    speedConnection = RunService.Stepped:Connect(function()

        if not success and speedConnection then
            speedConnection:Disconnect()
            speedConnection = nil
        end
    end)
end

local function StartFly()
    if flyEnabled then return end
    flyEnabled = true
    swimMethod = true

    task.wait(0.2)
    
    local character = player.Character or player.CharacterAdded:Wait()
    if not character then return end

    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local head = character:FindFirstChild("Head")
    if not humanoid or not head then return end

    humanoid.PlatformStand = true
    head.Anchored = true

    if CFloop then 
        CFloop:Disconnect() 
    end

    CFloop = RunService.Heartbeat:Connect(function(deltaTime)
        if not flyEnabled or not character or not character.Parent then
            if CFloop then CFloop:Disconnect() end
            return
        end
        
        local moveDir = humanoid.MoveDirection
        local speed = (Config.FlySpeed or 5) * deltaTime * 100
        
        local camCF = camera.CFrame
        local _, yRot, _ = camCF:ToOrientation()
        local uprightRot = CFrame.Angles(0, yRot, 0)

        local flatCamCF = CFrame.new(camCF.Position) * uprightRot
        local localInput = flatCamCF:VectorToObjectSpace(moveDir * speed)
        local worldMovement = camCF:VectorToWorldSpace(localInput)

        head.CFrame = CFrame.new(head.Position + worldMovement) * uprightRot
        camera.CameraSubject = head
    end)
end

local function StopFly()
    if not flyEnabled then return end
    flyEnabled = false
    swimMethod = false

    if CFloop then
        CFloop:Disconnect()
        CFloop = nil
    end
    
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        local head = character:FindFirstChild("Head")
        
        if humanoid then
            humanoid.PlatformStand = false
            humanoid:ChangeState(Enum.HumanoidStateType.Running)
        end
        
        if head then
            head.Anchored = false
        end
    end
end

local function setupMovement()
	bodyGyro = Instance.new("BodyGyro")
	bodyGyro.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
	bodyGyro.P = 50000
	bodyGyro.D = 1000
	bodyGyro.CFrame = humanoidRootPart.CFrame
	bodyGyro.Parent = humanoidRootPart

	movementConnection = RunService.RenderStepped:Connect(function()
		if not enhancedWalk then return end

		local camera = workspace.CurrentCamera
		local dir = Vector3.zero
		local usingKeys = false

		if UserInputService:IsKeyDown(Enum.KeyCode.W) then
			dir += Vector3.new(camera.CFrame.LookVector.X, 0, camera.CFrame.LookVector.Z)
			usingKeys = true
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.S) then
			dir -= Vector3.new(camera.CFrame.LookVector.X, 0, camera.CFrame.LookVector.Z)
			usingKeys = true
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.A) then
			dir -= camera.CFrame.RightVector
			usingKeys = true
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.D) then
			dir += camera.CFrame.RightVector
			usingKeys = true
		end

		if not usingKeys then
			local md = humanoid.MoveDirection
			dir = Vector3.new(md.X, 0, md.Z)
		end

		local moveSpeed = speed
		if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
			moveSpeed = moveSpeed * boostMultiplier
		end

		if dir.Magnitude > 0 then
			dir = dir.Unit * moveSpeed
		end

		local currentY = humanoidRootPart.AssemblyLinearVelocity.Y
		local groundedY = math.clamp(currentY, -100, -2)
		humanoidRootPart.AssemblyLinearVelocity = Vector3.new(dir.X, groundedY, dir.Z)

		if dir.Magnitude > 0 then
			local look = dir.Unit
			bodyGyro.CFrame = CFrame.new(humanoidRootPart.Position, humanoidRootPart.Position + Vector3.new(look.X, 0, look.Z))
		end

		humanoidRootPart.RotVelocity = Vector3.zero
		humanoidRootPart.AssemblyAngularVelocity = Vector3.zero

		if dir.Magnitude > 0 then
			if not animationTrack.IsPlaying then animationTrack:Play() end
		else
			if animationTrack.IsPlaying then animationTrack:Stop() end
		end
	end)
end

local function applySky(choice)
    -- Handle Compkiller dropdown passing a table
    if type(choice) == "table" then
        choice = choice.Value or choice[1] or tostring(choice)
    end
    choice = tostring(choice)

    if choice == "None" then
        pcall(function()
            if originalSettings.Brightness then Lighting_Sky.Brightness = originalSettings.Brightness end
            if originalSettings.ClockTime then Lighting_Sky.ClockTime = originalSettings.ClockTime end
            if originalSettings.FogEnd then Lighting_Sky.FogEnd = originalSettings.FogEnd end
            if originalSettings.FogColor then Lighting_Sky.FogColor = originalSettings.FogColor end
            if originalSettings.GlobalShadows then Lighting_Sky.GlobalShadows = originalSettings.GlobalShadows end
            if originalSettings.OutdoorAmbient then Lighting_Sky.OutdoorAmbient = originalSettings.OutdoorAmbient end
            if originalSettings.Ambient then Lighting_Sky.Ambient = originalSettings.Ambient end
            if originalSettings.EnvironmentDiffuseScale then Lighting_Sky.EnvironmentDiffuseScale = originalSettings.EnvironmentDiffuseScale end
            if originalSettings.EnvironmentSpecularScale then Lighting_Sky.EnvironmentSpecularScale = originalSettings.EnvironmentSpecularScale end
            if originalSettings.ExposureCompensation then Lighting_Sky.ExposureCompensation = originalSettings.ExposureCompensation end
        end)
        return
    end

    local preset = skyPresets[choice]
    if not preset then return end

    pcall(function()
        Lighting_Sky.ClockTime = preset.ClockTime
        Lighting_Sky.Brightness = preset.Brightness
        Lighting_Sky.FogEnd = preset.FogEnd
        Lighting_Sky.FogColor = preset.FogColor
        Lighting_Sky.GlobalShadows = preset.GlobalShadows
        Lighting_Sky.OutdoorAmbient = preset.OutdoorAmbient
        Lighting_Sky.Ambient = preset.Ambient
        Lighting_Sky.EnvironmentDiffuseScale = preset.EnvironmentDiffuseScale
        Lighting_Sky.EnvironmentSpecularScale = preset.EnvironmentSpecularScale
        Lighting_Sky.ExposureCompensation = preset.ExposureCompensation

        -- Update or create Atmosphere
        local atm = Lighting_Sky:FindFirstChildOfClass("Atmosphere")
        if not atm and preset.Atmosphere then
            atm = Instance.new("Atmosphere")
            atm.Parent = Lighting_Sky
        end
        if atm then
            atm.Color = preset.AtmColor
            atm.Decay = preset.AtmDecay
            atm.Glare = preset.AtmGlare
            atm.Haze = preset.AtmHaze
        end
    end)
end

local CarFly = { Enabled = false, Speed = 150 }

local function StopCarFly()
    CarFly.Enabled = false
    if flyConnection then
        flyConnection:Disconnect()
        flyConnection = nil
    end
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name == "CASH APP SCRIPTSCarVelocity" or v.Name == "CASH APP SCRIPTSCarGyro" then
            v:Destroy()
        end
    end

local function StartCarFly()
    local char = player.Character
    if not char then return end
    local humanoid = char:FindFirstChildWhichIsA("Humanoid")
    if not humanoid then return end
    local seat = humanoid.SeatPart
    if not seat or not seat:IsA("VehicleSeat") then
        WindUI:Notify({ Title = "Car Fly", Content = "Sit in a vehicle first!", Duration = 3 })
        return
    end
    local root = seat.Parent.PrimaryPart or seat
    StopCarFly()
    CarFly.Enabled = true
    local bv = Instance.new("BodyVelocity")
    bv.Name = "CASH APP SCRIPTSCarVelocity"
    bv.MaxForce = Vector3.new(1e9,1e9,1e9)
    bv.Parent = root
    local bg = Instance.new("BodyGyro")
    bg.Name = "CASH APP SCRIPTSCarGyro"
    bg.MaxTorque = Vector3.new(1e9,1e9,1e9)
    bg.P = 50000
    bg.D = 1000
    bg.CFrame = root.CFrame
    bg.Parent = root
    flyConnection = RunService.RenderStepped:Connect(function()
        if not CarFly.Enabled then StopCarFly() return end
        if not seat.Parent then StopCarFly() return end
        local cam = workspace.CurrentCamera
        local moveDir = Vector3.zero
        local look = cam.CFrame.LookVector
        local right = cam.CFrame.RightVector
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir += look end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir -= look end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir -= right end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir += right end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir += Vector3.new(0,1,0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir -= Vector3.new(0,1,0) end
        if moveDir.Magnitude > 0 then
            bv.Velocity = moveDir.Unit * CarFly.Speed
            bg.CFrame = CFrame.new(root.Position, root.Position + moveDir)
        else
            bv.Velocity = Vector3.zero
        end
    end)
end

-- ==================== IMPROVED BRING NEAREST CAR ====================
local function GetNearestCar(hrp)
    local CivCars = workspace:FindFirstChild("CivCars")
    if not CivCars then return nil end
    local nearestCar = nil
    local nearestDist = math.huge
    for _, car in ipairs(CivCars:GetChildren()) do
        if not car:IsA("Model") then continue end
        local seat = car:FindFirstChild("DriveSeat") or car:FindFirstChildWhichIsA("VehicleSeat")
        if not seat or seat.Occupant then continue end
        if not car.PrimaryPart then
            car.PrimaryPart = car:FindFirstChildWhichIsA("BasePart")
        end
        if not car.PrimaryPart then continue end
        local dist = (car.PrimaryPart.Position - hrp.Position).Magnitude
        if dist < nearestDist then
            nearestDist = dist
            nearestCar = car
        end
    end
    return nearestCar
end

-- Continuously apply all gun mods to equipped weapon
local function applyGunMods()
    local char = LocalPlayer.Character
    if not char then return end
    local tool = char:FindFirstChildOfClass("Tool")
    if not tool or not tool:FindFirstChild("Setting") then return end

    pcall(function()
        local setting = require(tool.Setting)

        if GunMods.InfiniteAmmo then
            setting.Ammo = 99999
            setting.AmmoPerMag = 99999
            setting.MaxAmmo = 99999
            setting.LimitedAmmoEnabled = false
        end
        if GunMods.Bullets80k then
            setting.Ammo = 80000
            setting.AmmoPerMag = 80000
            setting.MaxAmmo = 80000
            setting.LimitedAmmoEnabled = false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000)
                        debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if GunMods.InstantReload then
            setting.ReloadTime = 0
        end
        if GunMods.InstantEquip then
            setting.EquipTime = 0
            setting.EquippingTime = 0
        end
        if GunMods.Unjam then
            setting.JamChance = 0
        end
        if GunMods.NoRecoil then
            setting.Recoil = 0
            setting.CameraRecoilingEnabled = false
        end
        if GunMods.Automatic then
            setting.Auto = true
        end
        if GunMods.NoFireRate then
            setting.FireRate = 0
        end
        if GunMods.InfiniteDamage then
            setting.BaseDamage = 9e9
        end
        if GunMods.NoSpread then
            setting.Spread = 0
            setting.SpreadX = 0
            setting.SpreadY = 0
            setting.Accuracy = 1
        end
        if GunMods.Flamethrower then
            setting.FireRate = 0
            setting.Auto = true
            setting.Spread = 0
            setting.BaseDamage = math.max(setting.BaseDamage or 10, 15)
            -- Enable fire/ignite effect on the gun
            local gunServer = tool:FindFirstChild("GunScript_Server")
            if gunServer then
                local ignite = gunServer:FindFirstChild("IgniteScript")
                if ignite then
                    -- Force ignite on every shot
                    setting.Ignite = true
                    setting.IgniteScript = ignite
                    setting.IgniteChance = 1
                    setting.IgniteDuration = 5
                    setting.IgniteDamage = 10
                end
            end
            -- Make it shoot like a flamethrower (multiple bullets per shot)
            setting.ShotgunEnabled = true
            setting.Pellets = 12
            setting.SpreadX = 0.3
            setting.SpreadY = 0.3
        end
    end)
end

function checkgun()
    local player = game.Players.LocalPlayer
    local gunTool = nil

    for _, v in pairs(player.Backpack:GetDescendants()) do
        if v:IsA("LocalScript") and v.Name == "GunScript_Local" then
            gunTool = v.Parent
            break
        end
    end

    if not gunTool and player.Character then
        for _, v in pairs(player.Character:GetDescendants()) do
            if v:IsA("LocalScript") and v.Name == "GunScript_Local" then
                gunTool = v.Parent
                break
            end
        end
    end

    if gunTool and gunTool:IsA("Tool") then
        player.Character:WaitForChild("Humanoid"):EquipTool(gunTool)
    end

    return gunTool
end

    local t = Instance.new("TextLabel")

local function CS_ESP_Create(plr)
    if plr == CS_ESP_LocalPlayer or CS_ESP_Objects[plr] then return end

    local holder = Instance.new("Folder")
    holder.Name = "ESP_" .. plr.Name
    holder.Parent = CS_ESP_Gui

    local box = {}
    for i = 1, 8 do
        box[i] = CS_ESP_NewLine(holder)
    end

    local distance = CS_ESP_NewText(holder)
    local weapon = CS_ESP_NewText(holder)
    local name = CS_ESP_NewText(holder)
    local healthBack = Instance.new("Frame")
    healthBack.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
    healthBack.BorderSizePixel = 0
    healthBack.Visible = false
    healthBack.Parent = holder

    local healthFill = Instance.new("Frame")
    healthFill.BackgroundColor3 = Color3.fromRGB(170, 170, 170)
    healthFill.BorderSizePixel = 0
    healthFill.Parent = healthBack

    local highlight = Instance.new("Highlight")
    highlight.Name = "CompServices_Chams"
    highlight.FillColor = CS_ESP_Settings.ChamFill
    highlight.OutlineColor = CS_ESP_Settings.Accent
    highlight.FillTransparency = 0.55
    highlight.OutlineTransparency = 0
    highlight.Enabled = false
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Parent = CS_ESP_CoreGui

    CS_ESP_Objects[plr] = {
        Holder = holder,
        Box = box,
        Distance = distance,
        Weapon = weapon,
        Name = name,
        HealthBack = healthBack,
        HealthFill = healthFill,
        Highlight = highlight,
    }
end

local function CS_ESP_Remove(plr)
    local obj = CS_ESP_Objects[plr]
    if not obj then return end
    pcall(function() obj.Holder:Destroy() end)
    pcall(function() obj.Highlight:Destroy() end)
    CS_ESP_Objects[plr] = nil
end

local function CS_ESP_SetBox(box, x, y, w, h)
    local thick = 2
    local len = math.clamp(math.floor(w * 0.25), 8, 28)

    box[1].Position = UDim2.fromOffset(x, y)
    box[1].Size = UDim2.fromOffset(len, thick)
    box[2].Position = UDim2.fromOffset(x, y)
    box[2].Size = UDim2.fromOffset(thick, len)

    box[3].Position = UDim2.fromOffset(x + w - len, y)
    box[3].Size = UDim2.fromOffset(len, thick)
    box[4].Position = UDim2.fromOffset(x + w - thick, y)
    box[4].Size = UDim2.fromOffset(thick, len)

    box[5].Position = UDim2.fromOffset(x, y + h - thick)
    box[5].Size = UDim2.fromOffset(len, thick)
    box[6].Position = UDim2.fromOffset(x, y + h - len)
    box[6].Size = UDim2.fromOffset(thick, len)

    box[7].Position = UDim2.fromOffset(x + w - len, y + h - thick)
    box[7].Size = UDim2.fromOffset(len, thick)
    box[8].Position = UDim2.fromOffset(x + w - thick, y + h - len)
    box[8].Size = UDim2.fromOffset(thick, len)
end

local function CS_ESP_GetWeapon(char, plr)
    local tool = char and char:FindFirstChildOfClass("Tool")
    if tool then return tool.Name end
    local bp = plr:FindFirstChildOfClass("Backpack")
    if bp then
        local backpackTool = bp:FindFirstChildOfClass("Tool")
        if backpackTool then
            return backpackTool.Name
        end
    end
    return "None"
end

    obj.Distance.Visible = false
    obj.Weapon.Visible = false
    obj.Name.Visible = false 
    obj.HealthBack.Visible = false
    obj.Highlight.Enabled = false
end

    CS_ESP_Camera = workspace.CurrentCamera

    for _, plr in ipairs(CS_ESP_Players:GetPlayers()) do
        if plr ~= CS_ESP_LocalPlayer then
            CS_ESP_Create(plr)
            local obj = CS_ESP_Objects[plr]
            local char = plr.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if not CS_ESP_Settings.Enabled or not char or not root or not hum or hum.Health <= 0 then
                CS_ESP_Hide(obj)
                continue
            end

            local dist = (CS_ESP_Camera.CFrame.Position - root.Position).Magnitude
            if dist > CS_ESP_Settings.MaxDistance then
                CS_ESP_Hide(obj)
                continue
            end

            local rootPos, onScreen = CS_ESP_Camera:WorldToViewportPoint(root.Position)
            if not onScreen then
                CS_ESP_Hide(obj)
                continue
            end

            local scale = math.clamp(1 / (rootPos.Z * math.tan(math.rad(CS_ESP_Camera.FieldOfView / 2)) * 2) * 1000, 0, 1000)
            local width = math.floor(4 * scale)
            local height = math.floor(6 * scale)
            local x = math.floor(rootPos.X - width / 2)
            local y = math.floor(rootPos.Y - height / 2)

            for _, line in ipairs(obj.Box) do
                line.BackgroundColor3 = CS_ESP_Settings.Accent
                line.Visible = CS_ESP_Settings.CornerBoxes
            end
            if CS_ESP_Settings.CornerBoxes then
                CS_ESP_SetBox(obj.Box, x, y, width, height)
            end

            obj.Highlight.Adornee = char
            obj.Highlight.FillColor = CS_ESP_Settings.ChamFill
            obj.Highlight.OutlineColor = CS_ESP_Settings.Accent
            obj.Highlight.Enabled = CS_ESP_Settings.Chams

obj.Name.Position = UDim2.fromOffset(x + width / 2 - 75, y - 32)
obj.Name.Size = UDim2.fromOffset(150, 16)
obj.Name.Visible = CS_ESP_Settings.Names

-- Helper: get good cleaner
local function GetGoodCleaner()
    if not workspace:FindFirstChild("1# Map") then return nil end
    local CounterInstance
    for _, Value in pairs(workspace["1# Map"]:GetChildren()) do
        if Value:FindFirstChild("CounterM") then
            CounterInstance = Value
            break
        end
    end
    if not CounterInstance then return nil end
    for _, _Value in pairs(CounterInstance:GetChildren()) do
        local cashPrompt = _Value:FindFirstChild("CashPrompt", true)
        local grabPrompt = _Value:FindFirstChild("GrabPrompt", true)
        if cashPrompt and cashPrompt.Enabled and cashPrompt.ObjectText == "Count Bread" and grabPrompt and not grabPrompt.Enabled then
            return _Value
        end
    end
    return nil
end

-- Drop money amount
local function DropMoneyAmount(Amount)
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local BankRemote = ReplicatedStorage:FindFirstChild('BankProcessRemote', true)
    if not BankRemote then return false end

    local function send(...)
        if BankRemote.InvokeServer then return BankRemote:InvokeServer(...)
        elseif BankRemote.FireServer then return BankRemote:FireServer(...) end
    end

    local attempts = {
        {"Drop", tostring(Amount)},
        {"Drop", tonumber(Amount) or Amount},
        {{"Drop", tostring(Amount)}},
        {{"Drop", tonumber(Amount) or Amount}},
    }
    for _, args in ipairs(attempts) do
        local ok = pcall(send, table.unpack(args))
        if ok then return true end
    end
    return false
end

-- Auto deposit
local function autoDepositAmount(amount)
    pcall(function()
        local BankAction = game:GetService("ReplicatedStorage"):FindFirstChild("BankAction")
        if BankAction then
            if amount and tonumber(amount) and tonumber(amount) > 0 then
                BankAction:FireServer("depo", tostring(math.floor(amount)))
            else
                BankAction:FireServer("depo", tostring(999999999))
            end
        end
    end)
end

-- Format money
local function formatMoney(amount)
    local formatted = tostring(math.floor(tonumber(amount) or 0))
    while true do
        local newFormatted, count = formatted:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        formatted = newFormatted
        if count == 0 then break end
    end
    return "$" .. formatted
end

-- Draggable phone
local phoneDragging, phoneDragStart, phoneStartPos
phoneChassis.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        phoneDragging = true
        phoneDragStart = input.Position
        phoneStartPos = phoneChassis.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                phoneDragging = false
            end
        end)
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if phoneDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - phoneDragStart
        phoneChassis.Position = UDim2.new(
            phoneStartPos.X.Scale, phoneStartPos.X.Offset + delta.X,
            phoneStartPos.Y.Scale, phoneStartPos.Y.Offset + delta.Y
        )
    end
end)

-- Keyboard shortcut: RightShift to toggle
game:GetService("UserInputService").InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == Enum.KeyCode.RightShift or input.KeyCode == Enum.KeyCode.Insert then
        toggleOregonUI()
    end
end)
end)
