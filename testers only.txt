local function formatMoney(amount)
    local f = tostring(math.floor(tonumber(amount) or 0))
    while true do local n, c = f:gsub("^(-?%d+)(%d%d%d)", "%1,%2"); f = n; if c == 0 then break end end
    return "$" .. f
end

local function duplicateCurrentItem()
    if DupeCooldown then return end
    DupeCooldown = true
    local char = LocalPlayer.Character
    local backpack = LocalPlayer:FindFirstChild("Backpack")
    if not char or not backpack then DupeCooldown = false; return end
    local tool = char:FindFirstChildOfClass("Tool")
    if not tool then DupeCooldown = false; return end
    local toolName = tool.Name
    local toolId = nil
    tool.Parent = backpack; task.wait(0.5)
    local conn
    if ReplicatedStorage:FindFirstChild("MarketItems") then
        conn = ReplicatedStorage.MarketItems.ChildAdded:Connect(function(item)
            if item.Name == toolName then
                local owner = item:WaitForChild("owner", 2)
                if owner and owner.Value == LocalPlayer.Name then toolId = item:GetAttribute("SpecialId") end
            end
        end)
    end
    task.spawn(function() local r = ReplicatedStorage:FindFirstChild("ListWeaponRemote"); if r then r:FireServer(toolName, 676767) end end)
    task.wait(0.25 + (math.clamp(getPing(), 0, 300) / 300) * 0.03)
    task.spawn(function() local r = ReplicatedStorage:FindFirstChild("BackpackRemote"); if r then r:InvokeServer("Store", toolName) end end)
    task.wait(3)
    if toolId then task.spawn(function() local r = ReplicatedStorage:FindFirstChild("BuyItemRemote"); if r then r:FireServer(toolName, "Remove", toolId) end end) end
    task.spawn(function() local r = ReplicatedStorage:FindFirstChild("BackpackRemote"); if r then r:InvokeServer("Grab", toolName) end end)
    task.wait(1)
    local duped = backpack:FindFirstChild(toolName)
    if duped then duped.Parent = char end
    if conn then conn:Disconnect() end
    task.wait(2); DupeCooldown = false
end

task.spawn(function()
    while true do
        task.wait(0.5)
        if getgenv().AutoDupeCurrentItem and not DupeCooldown then pcall(duplicateCurrentItem) end
    end
end)

local function GetPlayerMoneyAmount()
    local pl = LocalPlayer
    if not pl then return nil end
    local leader = pl:FindFirstChild("leaderstats") or pl:FindFirstChild("Leaderstats") or pl:FindFirstChild("stats")
    if leader then
        for _, name in ipairs({"Money","Cash","Bank","Coins","Wallet"}) do
            local val = leader:FindFirstChild(name)
            if val and type(val.Value) == "number" then return val.Value end
        end
    end
    if pl.GetAttribute then
        local a = pl:GetAttribute("Money") or pl:GetAttribute("Cash") or pl:GetAttribute("Coins")
        if a and tonumber(a) then return tonumber(a) end
    end
    return nil
end

local function autoDepositAmount(amount)
    pcall(function()
        local BankAction = ReplicatedStorage:FindFirstChild("BankAction")
        if BankAction then
            if amount and tonumber(amount) and tonumber(amount) > 0 then
                BankAction:FireServer("depo", tostring(math.floor(amount)))
            else
                BankAction:FireServer("depo", tostring(999999999))
            end
        end
    end)
end

local function DropMoneyAmount(Amount)
    local BankRemote = ReplicatedStorage:FindFirstChild('BankProcessRemote', true)
    if not BankRemote then return end
    local function send(...)
        if BankRemote.InvokeServer then return BankRemote:InvokeServer(...)
        elseif BankRemote.FireServer then return BankRemote:FireServer(...) end
    end
    pcall(send, "Drop", tostring(Amount))
end

        if Value:FindFirstChild("CounterM") then CounterInstance = Value; break end

    for _, _Value in pairs(CounterInstance:GetChildren()) do

        if GunMods.InstantEquip then s.EquipTime = 0; s.EquippingTime = 0 end
        if GunMods.Unjam then s.JamChance = 0 end

-- ========== HITBOX EXPANDER ==========

getgenv().HitboxColor = Color3.fromRGB(0, 150, 255)
local OriginalHitboxProps = {}

local function restoreHitbox(part)
    local props = OriginalHitboxProps[part]
    if props and part and part.Parent then
        pcall(function()
            part.Size = props.Size
            part.Transparency = props.Transparency
            part.Massless = props.Massless
            part.CanCollide = props.CanCollide
            part.CanTouch = props.CanTouch
            part.Color = props.Color
            part.Material = props.Material
            local sbox = part:FindFirstChild("HitboxSelectionBox")
            if sbox then sbox:Destroy() end
        end)
    end
end

task.spawn(function()
    while task.wait(0.1) do
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character then
                local char = p.Character
                local humanoid = char:FindFirstChildOfClass("Humanoid")
                
                if humanoid then
                    local choice = getgenv().HitboxPartChoice or "Head"
                    local targetPart = char:FindFirstChild(choice) or char:FindFirstChild("Head") or char.PrimaryPart
                    
                    -- If we switched parts or disabled, restore all other parts
                    for part, _ in pairs(OriginalHitboxProps) do
                        if part:IsDescendantOf(char) and (not getgenv().HitboxEnabled or part ~= targetPart or humanoid.Health <= 0 or humanoid.Sit) then
                            restoreHitbox(part)
                        end
                    end
                    
                    if getgenv().HitboxEnabled and targetPart and humanoid.Health > 0 and not humanoid.Sit then
                        if not OriginalHitboxProps[targetPart] then
                            OriginalHitboxProps[targetPart] = {
                                Size = targetPart.Size,
                                Transparency = targetPart.Transparency,
                                Massless = targetPart.Massless,
                                CanCollide = targetPart.CanCollide,
                                CanTouch = targetPart.CanTouch,
                                Color = targetPart.Color,
                                Material = targetPart.Material
                            }
                        end
                        
                        pcall(function()
                            local sz = getgenv().HitboxSize
                            targetPart.Size = Vector3.new(sz, sz, sz)
                            targetPart.Transparency = 0.5
                            targetPart.Massless = true
                            targetPart.CanCollide = false
                            targetPart.CanTouch = false
                            
                            if getgenv().HitboxColorEnabled then
                                targetPart.Material = Enum.Material.ForceField
                                targetPart.Color = getgenv().HitboxColor or Color3.fromRGB(0, 150, 255)
                            else
                                local orig = OriginalHitboxProps[targetPart]
                                if orig then
                                    targetPart.Material = orig.Material
                                    targetPart.Color = orig.Color
                                end
                            end
                            
                            local sbox = targetPart:FindFirstChild("HitboxSelectionBox")
                            if sbox then sbox:Destroy() end
                        end)
                    end
                end
            end
        end
    end
end)

local function getClosestSilent()

    local method = getnamecallmethod()
    if not checkcaller() and self.Name == "InflictTarget" and method == "FireServer" then
        if getgenv().SilentEmoteOnHit then
            task.spawn(function()
                pcall(function()
                    local choice = getgenv().SilentEmoteChoice or 98318652597666
                    playCustomEmote(choice)
                end)
            end)
        end
    end
    return RemoteInflictHook(self, ...)
end)

if hookmetamethod then
    pcall(function()
        local origNamecall
        origNamecall = hookmetamethod(workspace, "__namecall", function(self, ...)
            local method = getnamecallmethod()
            local lm = string.lower(method)
            if checkcaller() then return origNamecall(self, ...) end
            local isRaycast = string.find(lm, "raycast") ~= nil
            local isFindPartOnRay = string.find(lm, "findpartonray") ~= nil
            if not isRaycast and not isFindPartOnRay then return origNamecall(self, ...) end

            -- Only redirect gun raycasts, NOT camera or anything else
            local cs = getcallingscript and getcallingscript()
            if not cs then return origNamecall(self, ...) end
            local sn = cs.Name
            if sn ~= "GunScript_Local" and sn ~= "GunScript" and sn ~= "BulletVisualizerServerScript" and sn ~= "BulletVisualizerClientScript" then
                return origNamecall(self, ...)
            end

            if not getgenv().SilentEnabled then return origNamecall(self, ...) end
            if not SilentTarget or not SilentTarget.Character then return origNamecall(self, ...) end
            if not (math.random(1, 100) <= getgenv().SilentHitChance) then return origNamecall(self, ...) end

            local tp = SilentTarget.Character:FindFirstChild("Head") or SilentTarget.Character:FindFirstChild("HumanoidRootPart")
            if not tp then return origNamecall(self, ...) end

            local args = {...}
            local origin
            if isFindPartOnRay and args[1] and typeof(args[1]) == "Ray" then
                origin = args[1].Origin
            elseif isRaycast and args[1] then
                origin = args[1]
            end
            if not origin then return origNamecall(self, ...) end

            local direction = (tp.Position - origin).Unit * 9e17

            if isFindPartOnRay then
                triggerSilentHitEmote()
                if getgenv().SilentWallBang then return tp, tp.Position, Vector3.new(0, 0, 0) end
                args[1] = Ray.new(origin, direction)
                return origNamecall(self, unpack(args))
            elseif isRaycast then
                triggerSilentHitEmote()
                args[2] = direction
                return origNamecall(self, unpack(args))
            end
            return origNamecall(self, ...)
        end)
    end)
end

local flyEnabled = false

local function flyHandleSpeed()
    if not flySwimMethod then return end
    local char = LocalPlayer.Character
    if not char then return end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")

    if flySpeedConnection then flySpeedConnection:Disconnect(); flySpeedConnection = nil end
    pcall(function() char:WaitForChild("Humanoid", 5); char:WaitForChild("HumanoidRootPart", 5) end)
    flySpeedConnection = RunService.Stepped:Connect(function()
        local ok = pcall(flyHandleSpeed)
        if not ok and flySpeedConnection then flySpeedConnection:Disconnect(); flySpeedConnection = nil end
    end)
end

local function StartFly()
    if flyEnabled then return end
    flyEnabled = true; flySwimMethod = true; getgenv().SwimMethod = true
    task.wait(0.2)
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    if not char then return end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    local head = char:FindFirstChild("Head")
    if not humanoid or not head then return end
    humanoid.PlatformStand = true; head.Anchored = true
    if flyCFloop then flyCFloop:Disconnect() end
    flyCFloop = RunService.Heartbeat:Connect(function(dt)
        if not flyEnabled or not char or not char.Parent then if flyCFloop then flyCFloop:Disconnect() end return end
        local moveDir = humanoid.MoveDirection
        local speed = flySpeedVal * dt * 100
        local camCF = Camera.CFrame
        local _, yRot = camCF:ToOrientation()
        local uprightRot = CFrame.Angles(0, yRot, 0)
        local flatCamCF = CFrame.new(camCF.Position) * uprightRot
        head.CFrame = CFrame.new(head.Position + camCF:VectorToWorldSpace(flatCamCF:VectorToObjectSpace(moveDir * speed))) * uprightRot
        Camera.CameraSubject = head
    end)
end

local function StopFly()
    if not flyEnabled then return end
    flyEnabled = false; flySwimMethod = false; getgenv().SwimMethod = false
    if flyCFloop then flyCFloop:Disconnect(); flyCFloop = nil end
    local char = LocalPlayer.Character
    if char then
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        local head = char:FindFirstChild("Head")
        if humanoid then humanoid.PlatformStand = false; humanoid:ChangeState(Enum.HumanoidStateType.Running) end
        if head then head.Anchored = false end
    end
end

    carFlyConn = RunService.RenderStepped:Connect(function()

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir += look end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir -= look end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir -= right end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir += right end

        if moveDir.Magnitude > 0 then bv.Velocity = moveDir.Unit * getgenv().CarFlySpeed; bg.CFrame = CFrame.new(root.Position, root.Position + moveDir)

-- ========== KEEP TOOLS ON DEATH ==========
local KeepToolsOnDeath = false
local MarketProtectedTools = {}

local function FindMarketToolId(toolName)
    if not ReplicatedStorage:FindFirstChild("MarketItems") then return nil end
    for _, item in ipairs(ReplicatedStorage.MarketItems:GetChildren()) do
        if item.Name == toolName and item:FindFirstChild("owner") and item.owner.Value == LocalPlayer.Name then
            return item:GetAttribute("SpecialId") or item:GetAttribute("ToolId")
        end
    end
end

local function MarketProtectTool(toolName)
    if MarketProtectedTools[toolName] then return end
    local toolId, conn
    if ReplicatedStorage:FindFirstChild("MarketItems") then
        conn = ReplicatedStorage.MarketItems.ChildAdded:Connect(function(item)
            if item.Name == toolName and item:FindFirstChild("owner") and item.owner.Value == LocalPlayer.Name then
                toolId = item:GetAttribute("SpecialId")
            end
        end)
    end
    pcall(function() if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then LocalPlayer.Character.Humanoid:UnequipTools() end end)
    task.wait(0.15)
    pcall(function() local r = ReplicatedStorage:FindFirstChild("ListWeaponRemote"); if r then r:FireServer(toolName, 676767) end end)
    task.wait(0.3)
    pcall(function() local r = ReplicatedStorage:FindFirstChild("BackpackRemote"); if r then r:InvokeServer("Store", toolName) end end)
    task.wait(3)
    if not toolId then toolId = FindMarketToolId(toolName) end
    if conn then conn:Disconnect() end
    if toolId then MarketProtectedTools[toolName] = toolId end
end

local function ReleaseMarketTools()
    for toolName, toolId in pairs(MarketProtectedTools) do
        pcall(function()
            local r1 = ReplicatedStorage:FindFirstChild("BuyItemRemote"); if r1 then r1:FireServer(toolName, "Remove", toolId) end
            task.wait(0.5)
            local r2 = ReplicatedStorage:FindFirstChild("BackpackRemote"); if r2 then r2:InvokeServer("Grab", toolName) end
        end)
        MarketProtectedTools[toolName] = nil
    end
end

local function SetupKeepTools(char)
    if not char then return end
    local humanoid = char:WaitForChild("Humanoid")
    humanoid.Died:Connect(function()
        if KeepToolsOnDeath then local tool = char:FindFirstChildOfClass("Tool"); if tool then task.spawn(MarketProtectTool, tool.Name) end end
    end)
    humanoid.HealthChanged:Connect(function(value)
        if not KeepToolsOnDeath then return end
        if value > 0 and value <= 35 then
            local tool = char:FindFirstChildOfClass("Tool"); if tool then task.spawn(MarketProtectTool, tool.Name) end
        elseif value > 35 and next(MarketProtectedTools) then
            task.spawn(ReleaseMarketTools)
        end
    end)
end

if LocalPlayer.Character then SetupKeepTools(LocalPlayer.Character) end
LocalPlayer.CharacterAdded:Connect(function(char) SetupKeepTools(char); task.wait(0.5); if KeepToolsOnDeath then task.spawn(ReleaseMarketTools) end end)

    character.DescendantAdded:Connect(function(descendant)
        if NoKnockback and (descendant:IsA("BodyVelocity") or descendant:IsA("LinearVelocity") or descendant:IsA("VectorForce")) then
            task.wait()
            if descendant and descendant.Parent then descendant:Destroy() end
        end
    end)
end

if LocalPlayer.Character then SetupNoKnockback(LocalPlayer.Character) end
LocalPlayer.CharacterAdded:Connect(SetupNoKnockback)

local function refreshHouses()
    local houses = {}
    pcall(function()
        if workspace:FindFirstChild("HomeDoorSS") then
            for _, v in pairs(workspace.HomeDoorSS:GetChildren()) do
                if v:FindFirstChild("Owner") and v.Owner.Value and v.Owner.Value ~= "None" then
                    table.insert(houses, v.Owner.Value .. "'s House")
                end
            end
        end
    end)
    if #houses == 0 then table.insert(houses, "None") else table.sort(houses) end
    selectedHouse = houses[1]
    if houseDropdown and houseDropdown.Refresh then houseDropdown:Refresh(houses) end
    return houses
end

end})

local function smartBuyItem(itemName)
    pcall(function()
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        local oldCF = char.HumanoidRootPart.CFrame

        -- Helper to check folders
        local function scanFolder(folder)
            if not folder then return false end
            local entry = folder:FindFirstChild(itemName)
            if not entry then
                for _, child in pairs(folder:GetChildren()) do
                    if child.Name:lower():find(itemName:lower()) then entry = child; break end
                end
            end
            if entry then
                local prox = entry:FindFirstChildWhichIsA("ProximityPrompt", true)
                if prox then
                    local part = prox.Parent:IsA("BasePart") and prox.Parent or (entry:IsA("BasePart") and entry or entry.PrimaryPart)
                    if part then
                        Teleport(part.CFrame + Vector3.new(0, 2, 0))
                        task.wait(0.3)
                        for _ = 1, 10 do pcall(function() fireproximityprompt(prox) end); task.wait(0.1) end
                        Teleport(oldCF)
                        return true
                    end
                end
            end
            return false
        end

        if scanFolder(workspace:FindFirstChild("PromptPurchases")) then notify("Shop", "Purchased " .. itemName, 2); return end
        if scanFolder(workspace:FindFirstChild("GUNS")) then notify("Shop", "Purchased " .. itemName, 2); return end
        if scanFolder(workspace:FindFirstChild("BankItems")) then notify("Shop", "Purchased " .. itemName, 2); return end

        -- Remote invocations
        if itemName:find("Water") or itemName:find("Ice") or itemName == "Bagelite" or itemName:find("Lemon") then
            pcall(function() ReplicatedStorage:WaitForChild("ExoticShopRemote", 2):InvokeServer(itemName) end)
        end
        pcall(function() ReplicatedStorage:FindFirstChild("ShopRemote"):InvokeServer(itemName) end)
        notify("Shop", "Purchased " .. itemName, 2)
    end)
end

local EmotesSection = MiscPage:Section({Name = "Custom Emotes", Side = 2})

local loadedEmoteCache = {}
local currentEmoteTrack = nil

local function playCustomEmote(id)
    task.spawn(function()
        local char = LocalPlayer.Character
        if not char then return end
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        local animator = humanoid and humanoid:FindFirstChildOfClass("Animator")
        if not animator then return end
        
        if currentEmoteTrack then pcall(function() currentEmoteTrack:Stop() end) end
        
        local anim = loadedEmoteCache[id]
        if not anim then
            -- 1. Try GetObjects or InsertService to extract authentic Animation from UGC catalog item
            local objects = nil
            pcall(function() objects = game:GetObjects("rbxassetid://" .. tostring(id)) end)
            if not objects or not objects[1] then
                pcall(function()
                    local InsertService = game:GetService("InsertService")
                    local asset = InsertService:LoadLocalAsset("rbxassetid://" .. tostring(id)) or pcall(function() return InsertService:LoadAsset(id) end)
                    if asset then objects = type(asset) == "table" and asset or {asset} end
                end)
            end
            
            if objects and objects[1] then
                local obj = objects[1]
                if obj:IsA("Animation") then
                    anim = obj
                else
                    anim = obj:FindFirstChildWhichIsA("Animation", true)
                end
            end
            
            -- 2. Fallback basic Animation instance
            if not anim then
                anim = Instance.new("Animation")
                anim.AnimationId = "rbxassetid://" .. tostring(id)
            end
            
            loadedEmoteCache[id] = anim
        end

        -- 3. Play via Animator
        local success, _ = pcall(function()
            currentEmoteTrack = animator:LoadAnimation(anim)
            currentEmoteTrack.Priority = Enum.AnimationPriority.Action4
            currentEmoteTrack.Looped = true
            currentEmoteTrack:Play()
        end)
        
        -- 4. Fallback Animate script
        if not success or not currentEmoteTrack or not currentEmoteTrack.IsPlaying then
            pcall(function()
                local animate = char:FindFirstChild("Animate")
                if animate and animate:FindFirstChild("PlayEmote") then
                    anim.Name = "OhioCustomEmote"
                    animate.PlayEmote:Invoke(anim)
                end
            end)
        end
        
        notify("Emote", "Playing emote!", 2)
    end)
end
